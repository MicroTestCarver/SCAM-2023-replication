package pl.altkom.asc.lab.micronaut.poc.pricing.domain;

import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.Setter;
import pl.altkom.asc.lab.micronaut.poc.pricing.service.api.v1.commands.calculateprice.dto.QuestionAnswer;

import java.math.BigDecimal;
import java.util.Objects;

@Getter
@AllArgsConstructor
public class Cover {

    private String code;
    @Setter
    private BigDecimal price;


    @Override
    public boolean equals(Object obj){
        if (this == obj)
            return true;
        if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        Cover source = (Cover) obj;
        return Objects.equals(code, source.code) && Objects.equals(price, source.price) ;
    }
}
