package pl.altkom.asc.lab.micronaut.poc.pricing.service.api.v1.commands.calculateprice.dto;

import io.micronaut.core.annotation.Introspected;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

import java.util.Objects;

@Introspected
@NoArgsConstructor
@AllArgsConstructor
@Getter
@Setter
public class QuestionAnswer<T> {
    private String questionCode;
    private T answer;

    @Override
    public boolean equals(Object obj){
        if (this == obj)
            return true;
        if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        QuestionAnswer source = (QuestionAnswer) obj;
        return Objects.equals(questionCode, source.questionCode) && Objects.equals(answer, source.answer) ;
    }
}
